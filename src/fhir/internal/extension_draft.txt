      pub fn primitive_extension(
        field_name: String,
        existing_extension: List(Extension),
        cb: fn(List(Extension)) -> decode.Decoder(a),
      ) -> decode.Decoder(a) {
        let field_ext_name = \"_\" <> field_name
        decode.then(
          decode.map(
            decode.optionally_at(
              [field_ext_name, \"extension\"],
              [],
              decode.list(extension_decoder()),
            ),
            fn(field_ext) {
              case field_ext {
                [] -> existing_extension
                new_field_ext -> [
                  Extension(
                    id: None,
                    url: field_ext_name,
                    ext: ExtComplex(new_field_ext),
                  ),
                  ..existing_extension
                ]
              }
            },
          ),
          cb,
        )
      }

fn gen_copy_searchparams_from_sansio(first_entry: Entry) {
  let search_type =
    list.map(first_entry.resource.rest, fn(rest) {
      list.map(rest.resource, fn(res) {
        let name_capital = string.capitalise(res.type_)
        "pub type Sp"
        <> name_capital
        <> " = FHIRVERSION_sansio."
        <> "Sp"
        <> name_capital
        <> "\n"
      })
      |> string.concat
    })
    |> string.concat
  let search_type_new =
    list.map(first_entry.resource.rest, fn(rest) {
      list.map(rest.resource, fn(res) {
        let name_lower = string.lowercase(res.type_)
        "pub fn sp_"
        <> name_lower
        <> "_new(){FHIRVERSION_sansio.sp_"
        <> name_lower
        <> "_new()}\n"
      })
      |> string.concat
    })
    |> string.concat
  search_type <> search_type_new
}

pub fn us_core_ethnicity_to_json(us_core_ethnicity: UsCoreEthnicity) -> Json {
  let UsCoreEthnicity(id:, ombcategory:, detailed:, text:) = us_core_ethnicity
  let url = "http://hl7.org/fhir/us/core/StructureDefinition/us-core-ethnicity"
  let fields = [
    #("url", json.string(url)),
  ]
  let fields = case id {
    Some(v) -> [#("id", json.string(v)), ..fields]
    None -> fields
  }
  let ext_children = []
  let ext_children = [
    json.object([
      #("url", json.string("text")),
      #("valueString", json.string(text)),
    ]),
    ..ext_children
  ]
  let ext_children = case ombcategory {
    None -> ext_children
    Some(ombcategory) -> [
      json.object([
        #("url", json.string("ombCategory")),
        #("valueCoding", r4.coding_to_json(ombcategory)),
      ]),
      ..ext_children
    ]
  }
  let ext_children = case detailed {
    [] -> ext_children
    _ ->
      detailed
      |> list.map(fn(x) {
        json.object([
          #("url", json.string("detailed")),
          #("valueString", r4.coding_to_json(x)),
        ])
      })
      |> list.append(ext_children)
  }
  let fields = case ext_children {
    [] -> fields
    _ -> [#("extension", json.preprocessed_array(ext_children)), ..fields]
  }
  json.object(fields)
}
